// context/gameReducer.ts
import { EstadoJogo, CARTAS_INICIAIS } from '../types/game';
import { AcaoJogo } from './gameActions';

export function gameReducer(state: EstadoJogo, action: AcaoJogo): EstadoJogo {
    switch (action.type) {
        case 'INICIAR_JOGO':
            return iniciarJogo(action.jogadores);
        
        case 'JOGAR_CARTA':
            return jogarCarta(state, action.cartaId, action.jogadorId);
        
        case 'COMPRAR_CARTA':
            return comprarCarta(state, action.jogadorId);
        
        case 'PASSAR_TURNO':
            return passarTurno(state);
        
        case 'ADICIONAR_RECURSOS':
            return adicionarRecursos(state, action.jogadorId, action.recursos);
        
        default:
            return state;
    }
}

function iniciarJogo(nomesJogadores: string[]): EstadoJogo {
    const jogadores = nomesJogadores.map((nome, index) => ({
        id: `jogador-${index}`,
        nome,
        recursos: [
            { tipo: 'madeira', quantidade: 3 },
            { tipo: 'pedra', quantidade: 0 },
            { tipo: 'ouro', quantidade: 0 }
        ],
        mao: [CARTAS_INICIAIS[0]], // Cada jogador come√ßa com uma floresta
        pontos: 0
    }));

    return {
        jogadores,
        jogadorAtual: 0,
        baralho: [...CARTAS_INICIAIS],
        descarte: [],
        fase: 'acao',
        vencedor: null
    };
}

function jogarCarta(state: EstadoJogo, cartaId: string, jogadorId: string): EstadoJogo {
    const jogador = state.jogadores.find(j => j.id === jogadorId);
    if (!jogador) return state;

    const carta = jogador.mao.find(c => c.id === cartaId);
    if (!carta) return state;

    // Verificar se tem recursos suficientes
    const temRecursosSuficientes = carta.custo.every(custo => {
        const recursoJogador = jogador.recursos.find(r => r.tipo === custo.tipo);
        return recursoJogador && recursoJogador.quantidade >= custo.quantidade;
    });

    if (!temRecursosSuficientes) return state;

    // Remover custos
    const novosRecursos = [...jogador.recursos];
    carta.custo.forEach(custo => {
        const recursoIndex = novosRecursos.findIndex(r => r.tipo === custo.tipo);
        if (recursoIndex !== -1) {
            novosRecursos[recursoIndex].quantidade -= custo.quantidade;
        }
    });

    // Aplicar efeitos
    carta.efeito.forEach(efeito => {
        const recursoIndex = novosRecursos.findIndex(r => r.tipo === efeito.tipo);
        if (recursoIndex !== -1) {
            novosRecursos[recursoIndex].quantidade += efeito.quantidade;
        } else {
            novosRecursos.push({ ...efeito });
        }
    });

    // Atualizar jogador
    const jogadoresAtualizados = state.jogadores.map(j =>
        j.id === jogadorId
            ? {
                  ...j,
                  recursos: novosRecursos,
                  mao: j.mao.filter(c => c.id !== cartaId),
                  pontos: j.pontos + 1 // +1 ponto por carta jogada
              }
            : j
    );

    // Mover carta para o descarte
    const novoDescarte = [...state.descarte, carta];

    return {
        ...state,
        jogadores: jogadoresAtualizados,
        descarte: novoDescarte
    };
}

import { useGame } from '../context/GameContext';
import { Carta } from '../types/game';

export function useGameLogic() {
    const { state, dispatch } = useGame();

    const jogadorAtual = state.jogadores[state.jogadorAtual];

    const jogarCarta = (carta: Carta) => {
        if (!jogadorAtual) return;
        
        dispatch({
            type: 'JOGAR_CARTA',
            cartaId: carta.id,
            jogadorId: jogadorAtual.id
        });
    };

    const comprarCarta = () => {
        if (!jogadorAtual) return;
        
        dispatch({
            type: 'COMPRAR_CARTA',
            jogadorId: jogadorAtual.id
        });
    };

    const passarTurno = () => {
        dispatch({ type: 'PASSAR_TURNO' });
    };

    const iniciarJogo = (nomesJogadores: string[]) => {
        dispatch({
            type: 'INICIAR_JOGO',
            jogadores: nomesJogadores
        });
    };

    const verificarVitoria = (): string | null => {
        // Vit√≥ria por pontos (primeiro a chegar em 20 pontos)
        const vencedorPontos = state.jogadores.find(j => j.pontos >= 20);
        if (vencedorPontos) return vencedorPontos.id;

        // Vit√≥ria por recursos (primeiro a ter 10 ouros)
        const vencedorOuro = state.jogadores.find(j => {
            const ouro = j.recursos.find(r => r.tipo === 'ouro');
            return ouro && ouro.quantidade >= 10;
        });
        if (vencedorOuro) return vencedorOuro.id;

        return null;
    };

    return {
        state,
        jogadorAtual,
        jogarCarta,
        comprarCarta,
        passarTurno,
        iniciarJogo,
        verificarVitoria
    };
}
